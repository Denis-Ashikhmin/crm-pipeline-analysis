CREATE VIEW IF NOT EXISTS pipeline_enriched AS
SELECT
    sp.opportunity_id,
	sp.sales_agent,
	sp.product,
	sp.account,
	sp.deal_stage,
	sp.engage_date,
	sp.close_date,
	sp.close_value,
	a.sector,
	a.employees,
	a.revenue AS account_revenue,
	p.series,
	p.sales_price,
	st.manager,
	st.regional_office,
	ROUND(JULIANDAY(sp.close_date) - JULIANDAY(sp.engage_date), 1) AS cycle_days,
	ROUND(100.0 * (sp.close_value - p.sales_price)/NULLIF(p.sales_price, 0), 1) AS price_variance_pct,
	CASE WHEN sp.deal_stage IN('Won', 'Lost') THEN 'Closed'
	ELSE 'Open' END AS deal_status,
	CASE WHEN sp.engage_date<='2017-10-01' THEN 'Mature'
	ELSE 'Too recent' END AS cohort_maturity
FROM sales_pipeline sp
LEFT JOIN sales_teams st ON sp.sales_agent = st.sales_agent
LEFT JOIN accounts a    ON sp.account = a.account
LEFT JOIN products p    ON sp.product = p.product;
