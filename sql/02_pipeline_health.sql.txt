SELECT
    COUNT(*) AS total_deals,
	SUM(CASE WHEN deal_stage='Won' THEN 1 ELSE 0 END) AS total_won,
	SUM(CASE WHEN deal_stage='Lost' THEN 1 ELSE 0 END) AS total_lost,
	SUM(CASE WHEN deal_status='Open' THEN 1 ELSE 0 END) AS total_open,
	ROUND(100.0 * SUM(CASE WHEN deal_stage='Won' THEN 1 ELSE 0 END)/NULLIF(SUM(CASE WHEN deal_stage IN('Won', 'Lost') THEN 1 ELSE 0 END), 0), 1) AS win_rate_pct,
	ROUND(SUM(CASE WHEN deal_stage='Won' THEN close_value ELSE 0 END), 0) AS total_revenue,
	ROUND(AVG(CASE WHEN deal_stage='Won' THEN close_value END), 0) AS avg_deal_value,
	ROUND(SUM(CASE WHEN deal_stage='Won' THEN sales_price ELSE 0 END), 0) AS pipeline_value
	FROM pipeline_enriched;
