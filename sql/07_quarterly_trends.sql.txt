SELECT 
    strftime('%Y', close_date) || '-Q' ||
	CASE WHEN CAST(strftime('%m', close_date) AS INTEGER) BETWEEN 1 AND 3 THEN '1'
	WHEN CAST(strftime('%m', close_date) AS INTEGER) BETWEEN 4 AND 6 THEN '2'
	WHEN CAST(strftime('%m', close_date) AS INTEGER) BETWEEN 7 AND 9 THEN '3'
	WHEN CAST(strftime('%m', close_date) AS INTEGER) BETWEEN 10 AND 12 THEN '4' END AS quarter,
	COUNT(*) AS total_closed,
	SUM(CASE WHEN deal_stage='Won' THEN 1 ELSE 0 END) AS won,
	ROUND(100.0*SUM(CASE WHEN deal_stage='Won' THEN 1 ELSE 0 END)/NULLIF(COUNT(*), 0), 1) AS win_rate_pct,
	ROUND(SUM(CASE WHEN deal_stage='Won' THEN close_value ELSE 0 END), 0) AS quarterly_revenue,
	ROUND(AVG(CASE WHEN deal_stage='Won' THEN close_value END), 0) AS avg_deal_value
FROM pipeline_enriched
WHERE deal_status='Closed'
GROUP BY quarter
ORDER BY quarter;
